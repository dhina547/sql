✅ 1️⃣ PayDB – Transactions Table
USE PayDB;
GO

IF OBJECT_ID('dbo.Transactions', 'U') IS NULL
BEGIN
    CREATE TABLE dbo.Transactions (
        TxnID INT PRIMARY KEY,
        UserID INT,
        Amount DECIMAL(10,2),
        TxnDate DATE,
        Status VARCHAR(20)
    );
END
GO

IF NOT EXISTS (SELECT 1 FROM dbo.Transactions)
INSERT INTO dbo.Transactions VALUES

(102, 2, 750.00, '2026-02-01', 'FAILED'),
(103, 3, 1200.00, '2026-02-02', 'SUCCESS');
GO
